programa caramelo:
tabla entero -> mat1[3][3];
tabla entero -> mat2[3][3];
tabla entero -> matRes[3][3];
variable entero -> i,j,k;


funcion entero multiplyMat(){

  mat1[0][0] = 0;
  mat1[0][1] = 1;
  mat1[0][2] = 2;
  mat1[1][0] = 3;
  mat1[1][1] = 4;
  mat1[1][2] = 5;
  mat1[2][0] = 6;
  mat1[2][1] = 7;
  mat1[2][2] = 8;

  mat2[0][0] = 0;
  mat2[0][1] = 1;
  mat2[0][2] = 2;
  mat2[1][0] = 3;
  mat2[1][1] = 4;
  mat2[1][2] = 5;
  mat2[2][0] = 6;
  mat2[2][1] = 7;
  mat2[2][2] = 8;


  // multiplicacion matrices
  desde (i = 0) hasta (3) {
    desde (j = 0) hasta (3){
        matRes[i][j] = 0;
            desde (k = 0) hasta (3){
             matRes[i][j] = matRes[i][j]+mat1[i][k]*mat2[k][j];
        };
  };
};

//imprimir matriz respuesta
desde (i=0) hasta (3){
    desde (j = 0) hasta (3) {
    imprimir("Matriz respuesta:");
    imprimir(matRes[i][j]);
    };  
};

regresar (0);
};


inicio(){
